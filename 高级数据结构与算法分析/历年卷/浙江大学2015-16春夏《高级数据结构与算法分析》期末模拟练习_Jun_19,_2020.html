int Power(int N, int k){
    if(k==0) return 1;
    unsigned K = k;
    unsigned n = N;
    unsigned int result = 1;
    while(K>0){
        if(K%2){
            result = result*n%10007;
        }
        K /= 2;
        n = ((n%10007)*(n%10007))%10007;

    return result%10007;
}
